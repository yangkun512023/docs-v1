    # Quick start

    npx

    npx @vuepress-reco/theme-cli init my-blog

    npm

    # init
npm install @vuepress-reco/theme-cli -g
theme-cli init my-blog

# install
cd my-blog
npm install

# run
npm run dev

# build
npm run build

    yarn

    # init
yarn global add @vuepress-reco/theme-cli
theme-cli init my-blog

# install
cd my-blog
yarn install
